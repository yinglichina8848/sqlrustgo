## 5. 压力测试

### 5.1 并发压力测试

```rust
#[test]
fn test_concurrent_queries() {
    let rt = Runtime::new().unwrap();
    
    // 50 并发连接
    let handles: Vec<_> = (0..50).map(|_| {
        rt.spawn(async {
            for _ in 0..1000 {
                let result = execute_query("SELECT * FROM users").await;
                assert!(result.is_ok());
            }
        })
    }).collect();
    
    // 等待完成并统计
    let start = Instant::now();
    for handle in handles {
        rt.block_on(handle);
    }
    let duration = start.elapsed();
    
    // 50000 查询 / duration = QPS
    println!("QPS: {}", 50000.0 / duration.as_secs_f64());
}
```
