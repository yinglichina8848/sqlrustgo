#### 谓词下推 (Predicate Pushdown)

```rust
struct PredicatePushdownRule;

impl OptimizeRule for PredicatePushdownRule {
    fn apply(&self, plan: &mut LogicalPlan) -> Result<bool, SqlError> {
        match plan {
            LogicalPlan::Join { left, right, condition, .. } => {
                // 将 Join 条件拆分到子节点
                let (left_cond, right_cond, join_cond) = 
                    split_condition(condition, left.schema(), right.schema());
                
                if let Some(c) = left_cond {
                    *left = LogicalPlan::Filter {
                        input: Box::new(std::mem::replace(left, LogicalPlan::Empty).clone()),
                        predicate: c,
                    };
                }
                // ... 同样处理 right
                Ok(true)
            }
            _ => Ok(false),
        }
    }
}
```

#### 列裁剪 (Column Pruning)

```rust
struct ColumnPruningRule;

impl OptimizeRule for ColumnPruningRule {
    fn apply(&self, plan: &mut LogicalPlan) -> Result<bool, SqlError> {
        // 收集必需的列集合
        let required_cols = collect_required_columns(plan);
        
        match plan {
            LogicalPlan::Projection { input, exprs, .. } => {
                // 只保留必需的列
                let pruned_exprs: Vec<_> = exprs.iter()
                    .filter(|e| required_cols.contains(&e.name()))
                    .cloned()
                    .collect();
                *exprs = pruned_exprs;
                Ok(true)
            }
            _ => Ok(false),
        }
    }
}
```

### 4.3 成本模型

```rust
struct CostModel {
    // 成本参数
    seq_scan_cost: f64,
    index_scan_cost: f64,
    hash_join_cost: f64,
    sort_cost_per_row: f64,
}

impl CostModel {
    fn estimate_scan_cost(&self, table: &TableStats, has_index: bool) -> Cost {
        let row_count = table.row_count as f64;
        let row_size = table.avg_row_size as f64;
        
        if has_index {
            Cost::new(row_count * self.index_scan_cost)
        } else {
            Cost::new(row_count * row_size * self.seq_scan_cost)
        }
    }
    
    fn estimate_join_cost(&self, left: &Cost, right: &Cost, join_type: JoinType) -> Cost {
        match join_type {
            JoinType::Hash => Cost::new(left.value + right.value * HASH_JOIN_FACTOR),
            JoinType::NestedLoop => Cost::new(left.value * right.value),
            _ => Cost::new(left.value + right.value),
        }
    }
}
```

### 5.4 向量化实现

```rust
struct VectorizedFilter {
    predicate: Box<dyn VectorizedExpr>,
}

impl VectorizedFilter {
    fn filter(&self, batch: &RecordBatch) -> RecordBatch {
        // SIMD 优化的过滤
        let result = unsafe {
            // 使用 SIMD 指令并行处理
            simd_filter(
                batch.columns(),
                self.predicate.as_ref(),
                batch.num_rows(),
            )
        };
        
        // 根据 mask 选择行
        batch.filter(result)
    }
}

// 伪 SIMD 代码示例
unsafe fn simd_filter(columns: &[Array], predicate: &dyn VectorizedExpr, len: usize) -> Mask {
    let mut result = Mask::new(len);
    
    // 每次处理 256 位 (4x64bit)
    for chunk in columns.chunks_exact(4) {
        let values = load_unaligned(chunk.as_ptr());
        let mask = predicate.eval_simd(values);
        result.or(mask);
    }
    
    result
}
```

### 7.4 隔离级别实现

```rust
impl TransactionManager {
    pub fn get_visible_version(&self, key: &RowKey, tx_id: TxId) -> Option<RowVersion> {
        let isolation = self.get_isolation_level(tx_id);
        
        let versions = self.version_store.get(key)?;
        
        match isolation {
            IsolationLevel::ReadUncommitted => {
                // 返回最新版本（未提交也可读，可能脏读）
                versions.iter().find(|v| !v.is_deleted).cloned()
            }
            IsolationLevel::ReadCommitted => {
                // 返回已提交的最新版本
                versions.iter()
                    .filter(|v| v.end_ts.is_some() && v.end_ts <= tx_id)
                    .max_by_key(|v| v.end_ts)
                    .cloned()
            }
            IsolationLevel::RepeatableRead => {
                // 返回事务开始时的快照版本
                let tx_begin = self.transactions[&tx_id].start_ts;
                versions.iter()
                    .filter(|v| v.begin_ts <= tx_begin && v.end_ts > tx_begin)
                    .max_by_key(|v| v.begin_ts)
                    .cloned()
            }
            _ => unimplemented!(),
        }
    }
}
```

```rust
fn test_repeatable_read_snapshot() {

    let tx1 = mgr.begin(IsolationLevel::RepeatableRead);
```

```uml
      Optimizer -> Executor: PhysicalPlan
      Executor -> Executor: 执行计划
      alt 执行错误
        Executor --> Client: SqlError::ExecutionError
      else 成功
        Executor -> Storage: 读写数据
        alt 存储错误
          Storage --> Client: SqlError::StorageError
        else 成功
          Storage --> Executor: Result
          Executor --> Client: ExecutorResult
        end
      end
    end
  end
end

@enduml
```
