## 3. 集成测试

### 3.1 Storage 集成测试

```rust
#[cfg(test)]
mod storage_integration_tests {
    #[test]
    fn test_wal_replay() {
        let storage = create_test_storage();
        let wal = WALManager::new(temp_dir()).unwrap();
        
        // Write some data
        storage.insert("users", test_batch()).unwrap();
        wal.flush().unwrap();
        
        // Simulate crash
        drop(storage);
        let new_storage = create_test_storage();
        
        // Verify data is recovered
        let result = new_storage.scan(table_ref("users"));
        assert!(result.is_ok());
    }
    
    #[test]
    fn test_columnar_storage() {
        let storage = ColumnarStorage::new(temp_dir()).unwrap();
        let batch = create_test_batch(1000);
        
        storage.write_batch("orders", batch.clone()).unwrap();
        
        let result = storage.scan("orders", &[0, 1, 2]).unwrap();
        assert_eq!(result.num_rows(), 1000);
    }
}
```

### 3.2 Transaction 集成测试

```rust
#[cfg(test)]
mod transaction_integration_tests {
    #[test]
    fn test_mvcc_snapshot_isolation() {
        let tm = TransactionManager::new();
        
        let tx1 = tm.begin(IsolationLevel::Snapshot).unwrap();
        let tx2 = tm.begin(IsolationLevel::Snapshot).unwrap();
        
        // TX1 writes
        storage.write("key", "value1", tx1).unwrap();
        tm.commit(tx1).unwrap();
        
        // TX2 should see old value
        let value = storage.read("key", tx2).unwrap();
        assert_eq!(value, None); // or old value depending on impl
    }
    
    #[test]
    fn test_2pc_commit() {
        let coordinator = Coordinator::new();
        let participant1 = create_test_participant();
        let participant2 = create_test_participant();
        
        let txid = coordinator.begin_distributed(IsolationLevel::Serializable).unwrap();
        coordinator.add_participant(txid, participant1.id()).unwrap();
        coordinator.add_participant(txid, participant2.id()).unwrap();
        
        let result = coordinator.prepare(txid).unwrap();
        assert_eq!(result.vote, Vote::VoteCommit);
        
        coordinator.commit(txid).unwrap();
    }
}
```
