### 2.4 AccessControl

```rust
pub trait AccessControl: Send + Sync {
    fn check_permission(&self, user: &User, permission: &Permission) -> Result<bool>;
    fn check_table_access(&self, user: &User, table: &str, action: Action) -> Result<bool>;
    fn check_database_access(&self, user: &User, database: &str) -> Result<bool>;
}

pub struct RbacAccessControl {
    roles: RwLock<HashMap<RoleId, Arc<Role>>>,
    users: RwLock<HashMap<String, Arc<User>>>,
}

impl AccessControl for RbacAccessControl {
    fn check_permission(&self, user: &User, permission: &Permission) -> Result<bool> {
        let roles = self.roles.read().unwrap();
        for role_id in &user.roles {
            if let Some(role) = roles.get(role_id) {
                if role.permissions.contains(permission) {
                    return Ok(true);
                }
            }
        }
        Ok(false)
    }
}
```

                log_failures: true,

                min_length: 8,
                require_uppercase: true,

                min_age_days: 1,
                history_count: 5,
