```rust
impl Catalog {
    /// 验证 catalog 内部一致性
    pub fn check_invariants(&self) -> Result<(), CatalogError> {
        // 1. 检查所有表引用的列都存在
        for (schema_name, schema) in &self.schemas {
            for (table_name, table) in &schema.tables {
                self.validate_table_columns(schema_name, table_name, table)?;
                self.validate_foreign_keys(schema_name, table_name, table)?;
            }
        }

        // 2. 检查没有循环外键引用
        self.check_foreign_key_cycles()?;

        // 3. 检查主键列存在且唯一
        self.validate_primary_keys()?;

        Ok(())
    }

    fn validate_table_columns(&self, schema: &str, table: &str, tbl: &Table) -> Result<(), CatalogError> {
        let mut col_names: HashSet<&str> = HashSet::new();
        for col in &tbl.columns {
            if !col_names.insert(&col.name) {
                return Err(CatalogError::DuplicateColumn {
                    schema: schema.to_string(),
                    table: table.to_string(),
                    column: col.name.clone(),
                });
            }
        }
        Ok(())
    }
}
```

```rust
impl Catalog {
    /// 从存储重建 catalog
    pub fn rebuild(storage: &dyn StorageEngine) -> SqlResult<Self> {
        let mut catalog = Catalog::new();

        // 1. 扫描所有表
        for table_name in storage.list_tables()? {
            let info = storage.get_table_info(&table_name)?;

            // 2. 重建表元数据
            let columns = info.columns.iter().map(|col| {
                ColumnDefinition {
                    name: col.name.clone(),
                    data_type: parse_data_type(&col.data_type)?,
                    nullable: col.nullable,
                    is_unique: col.is_unique,
                    default_value: None,
                    references: col.references.clone(),
                }
            }).collect();

            let table = Table {
                name: table_name.clone(),
                columns,
                row_count: 0, // 从存储统计
                primary_key: info.primary_key.clone(),
                indices: Vec::new(),
            };

            catalog.get_schema_mut("public")?
                .add_table(table_name, Arc::new(table))?;
        }

        catalog.check_invariants()?;
        Ok(catalog)
    }
}
```

```rust
pub struct ExplainExecutor {
    physical_plan: Box<dyn PhysicalPlan>,
    verbose: bool,
}

impl ExplainExecutor {
    pub fn execute(&self) -> SqlResult<ExecutorResult> {
        let plan_desc = self.format_plan(self.physical_plan.as_ref(), 0);

        let rows = if self.verbose {
            self.format_verbose_plan()
        } else {
            vec![vec![Value::Text(plan_desc)]]
        };

        Ok(ExecutorResult {
            rows,
            affected_rows: 0,
        })
    }

    fn format_plan(&self, plan: &dyn PhysicalPlan, indent: usize) -> String {
        let prefix = "  ".repeat(indent);
        let mut s = format!("{}{:?}", prefix, plan.name());

        for child in plan.children() {
            s.push('\n');
            s.push_str(&self.format_plan(child.as_ref(), indent + 1));
        }

        s
    }
}
```

```rust
impl LocalExecutor {
```

```rust
impl Catalog {
    pub fn query_information_schema(&self, table_name: &str) -> SqlResult<Vec<Vec<Value>>> {
        match table_name {
            "TABLES" => self.query_tables(),
            "COLUMNS" => self.query_columns(),
            "SCHEMATA" => self.query_schemata(),
            _ => Err(SqlError::internal(format!(
                "Unknown INFORMATION_SCHEMA table: {}", table_name
            ))),
        }
    }

    fn query_tables(&self) -> SqlResult<Vec<Vec<Value>>> {
        let mut rows = Vec::new();

        for (schema_name, schema) in &self.schemas {
            for (table_name, table) in &schema.tables {
                rows.push(vec![
                    Value::Text("def".to_string()),           // table_catalog
```

```rust
impl LocalExecutor {
    pub fn with_telemetry(storage: S, telemetry: Arc<dyn Telemetry>) -> Self {
        Self { storage, telemetry }
    }

    pub fn execute(&self, plan: &dyn PhysicalPlan) -> SqlResult<ExecutorResult> {
        let query_id = Uuid::new_v4().to_string();

        self.telemetry.record_query_start(&query_id, "SELECT");
        let start = Instant::now();

        let result = self.execute_internal(plan);

        if let Ok(ref res) = result {
            self.telemetry.record_query_end(
                &query_id,
                start.elapsed().as_millis() as u64,
                res.rows.len() as u64,
            );
        }

        result
    }
}
```

```rust
    pub query_distribution: QueryDistribution,

}

#[tokio::test]
async fn test_72_hour_soak() {
    let config = SoakTestConfig {
        duration: Duration::hours(72),
        query_distribution: QueryDistribution {
            select: 0.7,
            insert: 0.2,
            update: 0.1,
        },
        concurrency: 10,
        data_size: DataSizeConfig { tables: 10, rows_per_table: 10000 },
    };

    let storage = create_test_storage();
    setup_test_data(&storage, &config.data_size)?;

    let executor = LocalExecutor::new(&storage);
    let start = Instant::now();
    let mut query_count = 0u64;
    let mut error_count = 0u64;

    while start.elapsed() < config.duration {
        let handles: Vec<_> = (0..config.concurrency).map(|_| {
            let executor = executor.clone();
            tokio::spawn(async move {
                let query = generate_random_query(&config.query_distribution);
                executor.execute(&query).await
            })
        }).collect();

        for handle in handles {
            match handle.await {
                Ok(Ok(())) => query_count += 1,
                Ok(Err(_)) => error_count += 1,
                Err(_) => error_count += 1,
            }
        }

        // 每小时报告进度
        if query_count % 10000 == 0 {
            println!("Progress: {} queries, {} errors", query_count, error_count);
        }
    }

    // 验证无错误或错误率 < 0.1%
    let error_rate = error_count as f64 / query_count as f64;
    assert!(error_rate < 0.001, "Error rate {} too high", error_rate);
}
```
