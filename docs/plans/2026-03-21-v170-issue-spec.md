pub struct UnionExec {
    left: Box<dyn PhysicalPlan>,
    right: Box<dyn PhysicalPlan>,
    schema: Schema,
}

impl PhysicalPlan for UnionExec {
    fn execute(&self) -> Result<Vec<Vec<Value>>, String> {
        let left_rows = self.left.execute()?;
        let right_rows = self.right.execute()?;
        if self.all {
            Ok([left_rows, right_rows].concat())
        } else {
            // DISTINCT - deduplicate
            let mut combined = [left_rows, right_rows].concat();
            combined.sort();
            combined.dedup();
            Ok(combined)
        }
    }
}
```

pub struct ExplainOptions {
    pub analyze: bool,

pub enum ExplainFormat {

}

pub fn explain_plan(
    plan: &dyn PhysicalPlan,
    options: &ExplainOptions,
) -> String {
    let mut output = String::new();
    explain_node(plan, 0, &mut output);
    output
}

fn explain_node(plan: &dyn PhysicalPlan, indent: usize, out: &mut String) {
    let prefix = "  ".repeat(indent);
    writeln!(out, "{}{}", prefix, plan.name());
    
    if let Some(cost) = plan.cost() {
        writeln!(out, "{}  cost={:.2}", prefix, cost);
    }
    
    for child in plan.children() {
        explain_node(child, indent + 1, out);
    }
}
```
