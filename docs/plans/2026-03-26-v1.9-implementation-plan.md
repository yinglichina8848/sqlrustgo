    pub fn execute_with_timing(&mut self, plan: &PhysicalPlan) -> SqlResult<(ExecutorResult, Duration)> {
        let start = Instant::now();
        let result = self.execute(plan)?;
        let duration = start.elapsed();
        Ok((result, duration))
    }
}
