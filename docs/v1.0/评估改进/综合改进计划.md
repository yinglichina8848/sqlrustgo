// src/executor/sort_executor.rs

pub struct SortExecutor {
    input: Box<dyn Executor>,
    order_by: OrderBy,
}

impl Executor for SortExecutor {
    fn execute(&mut self) -> SqlResult<ExecutionResult> {
        let mut rows = self.input.execute()?.into_rows();
        rows.sort_by(|a, b| {
            for order_expr in &self.order_by.expressions {
                let cmp = compare_values(&a[order_expr.index], &b[order_expr.index]);
                if order_expr.direction == OrderDirection::Desc {
                    return cmp.reverse();
                }
            }
            cmp
        });
        Ok(ExecutionResult::Rows(rows))
    }
}

        BufferPoolMetrics {

            misses: self.misses.load(Ordering::SeqCst),
            evictions: self.evictions.load(Ordering::SeqCst),

        let hits = self.hits.load(Ordering::SeqCst) as f64;
