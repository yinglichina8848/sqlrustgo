**技术方案**:

```rust
// Executor 修改
impl Executor {
    pub fn execute(&mut self, stmt: Statement) -> Result<Vec<Record>, Error> {
        let tx_id = self.mvcc.begin_transaction()?;
        let result = match stmt {
            Statement::Insert(ins) => self.execute_insert(ins, tx_id),
            // ...
        };
        if result.is_ok() {
            self.mvcc.commit_transaction(tx_id)?;
        } else {
            self.mvcc.abort_transaction(tx_id)?;
        }
        result
    }
}
```
